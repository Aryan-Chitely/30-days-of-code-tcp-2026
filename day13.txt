class Solution {
    public int minEatingSpeed(int[] piles, int h) {

        int low = 1;
        int high = getMaxPile(piles);
        int ans = high;

        while (low <= high) {
            int mid = (low+high)/2;

            if (canFinish(piles, h, mid)) {
                ans = mid;    
                high = mid-1;   
            } else {
                low = mid + 1;  
            }
        }

        return ans;
    }

    private int getMaxPile(int[] piles) {
        int max = 0;
        for (int pile : piles) {
            if (pile > max) {
                max = pile;
            }
        }
        return max;
    }

    boolean canFinish(int[] piles, long h, int mid) {
        long totalH = 0;

        for (int pile : piles) {
            totalH+=(pile+mid-1)/mid;
        }

        return totalH<=h;
    }
}
